1. 텍스트 전처리 (Preprocessing)¶

형태소 추출과 품사 처리에는 konlpy의 kkma(kkma.pos)를 사용할 것이며 자세한 내용은 http://konlpy.org에서 확인 가능하다. 감성사전은 http://word.snu.ac.kr/kosac/(서울대학교)를 사용할 것이다. 이 중 단어 이름/ 품사/ 긍정/ 중간/ 부정 의 열을 사용할 것이다.

Get_score(분류 함수, Classification function)¶

사전과 비교하여 긍정/ 부정으로 분류할 것이다.

In [80]:
def get_score(comment) : # 코멘트 하나가 들어간다.
    tmp = [''.join(cm) for cm in kkma.pos(comment)] # 코멘트(문장) 하나를 (단어, 형태소).. 로 잘라서 tmp 안에 저장한다. 
    # join은 ['단어', '형태소'] 같은 형태의 2개 이상의 문자열을 ''을 기준으로 합치는 것이다 -> '단어형태소'
    tmp_df = pd.DataFrame() # 빈 임시 데이터프레임을 만든다.
    
    for t in tmp : # tmp 안의 모든 내용에 대해 반복한다.
        if t in p_df.ngram.values : # 단어형태소(ex.영화NNG) 형태가 사전 안에 있으면
            tmp_df = pd.concat([tmp_df, p_df.query('ngram == "{}"'.format(t))]) # 해당 데이터프레임을 QUERY를 통해 추출하고 tmp_df에 추가한다.
            # concat 함수는 데이터프레임에 새로운 데이터프레임을 추가할 수 있게 해준다. (밑으로 합쳐짐)
            
    means = tmp_df.mean() # 한 코멘트에 있는 모든 단어들의 감성 점수의 평균을 낸다.
    
    if len(means) == 0 : # 빈 문장은 모두 0을 반환한다.
        means = {'POS' : 0, 'NEG' : 0, 'NEUT' : 0}
        return (means, 'NEUT')
    
    return (means, mean.index[np.argmax(mean.values)]) # 리뷰의 평균 확률을 반환한다.

def get_tp(clean_text, num_topics, num_words, lda_name = 'model.gensim', topics_name = 'corpus.pkl', dict_name = 'dictionary.gensim'):
    """
    clean_text : get_clean_txt로 전처리를 완료한 대륙 별 워드 리스트 (반드시 []로 감싸야한다.)
    num_topics : 출력할 토픽의 수
    num_words : 한 토픽에 포함될 단어의 수 
    ldamodel : lda 모델 세이브 네임
    topics_name : topic 저장할 세이브 네임
    dict_name : (값, 토픽) 저장할 세이브 네임
    """
    dictionary = corpora.Dictionary(clean_text)
    corpus = [dictionary.doc2bow(text) for text in clean_text]
    NUM_TOPICS = num_topics
    ldamodel = gensim.models.ldamodel.LdaModel(corpus, num_topics = NUM_TOPICS, id2word=dictionary, passes=1)
    ldamodel.save(lda_name)
    
    topics = ldamodel.print_topics(num_words=num_words)
    pickle.dump(corpus, open(topics_name, 'wb'))
    
    dictionary.save(dict_name)
    dictionary = gensim.corpora.Dictionary.load(dict_name)
    corpus = pickle.load(open(topics_name, 'rb'))
    lda = gensim.models.ldamodel.LdaModel.load(lda_name)    
    lda_display = pyLDAvis.gensim.prepare(lda, corpus, dictionary, sort_topics=False)
    pyLDAvis.display(lda_display)
    
    print(topics)
    return pyLDAvis.display(lda_display)
