def conv_load(target):
    if target in ['터널안','교량위','지하차도(도로)내','기타단일로','고가도로위']: # 비교적 높은 속도
        return 'H_speed'
    else: # 비교적 낮은 속도 혹은 교차로
        return 'L_speed_or_Intersection'
